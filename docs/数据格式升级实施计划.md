**正则表达式**:
```javascript
const TIME_FORMATS = {
  year:   { pattern: /^\d{4}$/,              name: '年度' },
  quarter:{ pattern: /^\d{4}Q[1-4]$/,        name: '季度' },
  month:  { pattern: /^\d{6}$/,              name: '月度' },
  week:   { pattern: /^\d{4}WK\d{2}$/,       name: '周度' },
  day:    { pattern: /^\d{8}$/,              name: '日报' }

```

**测试用例**:

- ✅ `2026Q1` → quarter
- ✅ `202601` → month
- ✅ `2026WK01` → week
- ✅ `20260101` → day
- ❌ `2026, 2026Q1, 202601` → 混合维度报错

### 2.2 Calculator 升级（4h）

**文件**: `src/engine/Calculator.js`

**修改内容**:

1. **新增 calculatePeriodValue 函数**:
```javascript
/**
 * 计算 PERIOD 函数值
 * @param {Object} node - 节点
 * @param {Object} nodes - 所有节点
 * @param {string[]} allNodeIds - 所有节点 ID
 * @param {string} attribute - 属性筛选：'AC'|'FC'|'BU'|''
 * @param {string} periodType - 周期类型：'D'|'W'|'M'|'Q'|'Y'
 */
function calculatePeriodValue(node, nodes, allNodeIds, attribute = '', periodType = '') {
  // 步骤 1：获取所有周期 keys
  const allKeys = Object.keys(node.originalTimeData);

  // 步骤 2：属性筛选
  let filteredKeys = allKeys;
  if (attribute === 'AC+FC' || attribute === '') {
    filteredKeys = allKeys.filter(k => k.endsWith('-AC') || k.endsWith('-FC'));
  } else {
    filteredKeys = allKeys.filter(k => k.endsWith(`-${attribute}`));
  }

  // 步骤 3：周期类型筛选
  if (periodType) {
    filteredKeys = filterByPeriodType(filteredKeys, periodType);
  }

  // 步骤 4：对每个周期计算
  const periodValues = [];
  filteredKeys.forEach(key => {
    periodValues.push(node.originalTimeData[key]);
  });

  // 步骤 5：汇总
  return sum(periodValues);
}
```

```javascript
export function aggregateTimeData(timeData = {}, aggregationType = 'sum') {
  const result = {
    actualTotal: 0,
    forecastTotal: 0,
    targetTotal: 0,
    actualPlusForecastTotal: 0,
    actualMonths: [],
    forecastMonths: [],
    targetMonths: [],
    actualCount: 0,
    forecastCount: 0,
    targetCount: 0
  };

  if (!timeData) return result;

  Object.entries(timeData).forEach(([key, value]) => {
    // 新格式：'202601-AC'
    const parts = key.split('-');
    if (parts.length !== 2) return;

    const [time, attr] = parts;
    const numValue = parseFloat(value);
    if (isNaN(numValue)) return;

    if (attr === 'AC') {
      result.actualTotal += numValue;
      result.actualMonths.push({ time, value: numValue });
      result.actualCount++;
    } else if (attr === 'FC') {
      result.forecastTotal += numValue;
      result.forecastMonths.push({ time, value: numValue });
      result.forecastCount++;
    } else if (attr === 'BU') {
      result.targetTotal += numValue;
      result.targetMonths.push({ time, value: numValue });
      result.targetCount++;
    }
  });

  // ... 后续逻辑不变
  return result;
}
```

### 6.2 逆运算引擎（2h）

**文件**: `src/engine/ReverseCalculation.js` (新建)

**功能**:
- 总额→分期分配
- 百分比调整
- 目标导向调整

**核心函数**:
```javascript
export function executeReverseCalculation(nodeId, totalAdjustment, options = {}) {
  const node = nodes[nodeId];
  const { distributionMode = 'proportional', weights = {}, periods = [] } = options;

  // 获取当前分期数据
  const periodData = {};
  periods.forEach(period => {
    const acKey = `${period}-AC`;
    const fcKey = `${period}-FC`;
    if (node.timeData[acKey] !== undefined) {
      periodData[period] = node.timeData[acKey];
    } else if (node.timeData[fcKey] !== undefined) {
      periodData[period] = node.timeData[fcKey];
    }
  });

  // 计算当前总额
  const currentTotal = Object.values(periodData).reduce((a, b) => a + b, 0);

  // 根据分配模式计算各期调整额
  let adjustmentMap = {};

  if (distributionMode === 'proportional') {
    Object.entries(periodData).forEach(([period, value]) => {
      const ratio = value / currentTotal;
      adjustmentMap[period] = totalAdjustment * ratio;
    });
  } else if (distributionMode === 'weight') {
    Object.entries(weights).forEach(([period, weight]) => {
      adjustmentMap[period] = totalAdjustment * weight;
    });
  }

  // 更新 timeData
  Object.entries(adjustmentMap).forEach(([period, adjustment]) => {
    const key = `${period}-AC`;
    if (node.timeData[key] !== undefined) {
      node.timeData[key] += adjustment;
    }
  });

  // 重新聚合 value
  recalculate();

  return { adjustmentMap, newTotal: currentTotal + totalAdjustment };
}
```
